function insertbillrow(tblElement) {

	var rowIndex = tblElement.parentNode.parentNode.rowIndex
	var cellIndex = tblElement.parentNode.cellIndex

	var table = document.getElementById("billaddressdetails");
	var rowCount = table.rows.length;


	try {

		var table = document.getElementById('billaddressdetails');
		var value = table.rows[rowIndex].cells[1]
				.getElementsByTagName('input')[0].value;
		;
		//var value = table.rows[rowIndex].cells[2].getElementsByTagName('select')[0].value;
		//add row
		if (rowIndex + 1 == rowCount) {
			var row = table.insertRow(rowCount);
			var colCount = table.rows[1].cells.length;
			for (var i = 0; i < colCount; i++) {

				var newcell = row.insertCell(i);

				if (i == 0) {
					table.rows[rowCount].cells[0].innerHTML = rowCount;
				} else {
					newcell.innerHTML = table.rows[1].cells[i].innerHTML;
				}

				//alert(newcell.childNodes);
				switch (newcell.childNodes[0].type) {
				case "text":
					newcell.childNodes[0].value = "";
					break;
				case "checkbox":
					newcell.childNodes[0].checked = false;
					break;
				case "select-one":
					newcell.childNodes[0].selectedIndex = 0;
					break;
				}
			}
		}

	} catch (e) {
		alert("Exceptation Occured!" + e);
	}
}
